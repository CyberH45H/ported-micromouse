41 /***
42  * Normally the systick event is triggered at a regular interval by a
43  * timer interrupt. In PORTEDmouse, the interval is 250 Hz. For a DC
44  * mouse with a faster processor, systick will run at 1kHz or faster
45  * and will also process the encoders and update all the PID controllers
46  * as well as generate the motor PWM signals
47  */
48 void systickInit(int frequency) {
49  timerReload = (F_CPU / 8L / frequency);
50  /// Timer 3 is used to generate the 500Hz system update
51  TCCR3A = 0x00; // normal port operation, no auto change of OCnA/B/C pins
52  TCNT3 = 0; // start the timer at 0x0000
53  OCR3A = timerReload;
54  TCCR3B = 0x02; // 0: off, 1 => FCPU/1, 2 => FCPU/8, 3 => FCPU/64
55  TIMSK3 |= (1 << OCIE3A);
56 }
57 
58 /***
59  * systick is defined as a function so that it may be called from
60  * test code when the interrupt is not running
61  */
62 
63 void systick() {
64  SREG |= (1 << SREG_I); // enable interrupts for the motor driver
65  navigatorUpdate();
66  debouncePin(BUTTON);
67  sensorUpdate();
68 }
