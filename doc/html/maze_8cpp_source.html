35 /***
36  * Set a single wall in the maze. Each wall is set from two directions
37  * so that it is consistent when seen from the neighbouring cell.
38  *
39  * The wall is set unconditionally regardless of whether there is
40  * already a wall present
41  *
42  * No check is made on the provided value for direction
43  */
44 void mazeSetWall(unsigned char cell, unsigned char direction) {
45  unsigned int nextCell = neighbour(cell, direction);
46  switch (direction) {
47  case NORTH:
48  walls[cell] |= (1 << NORTH);
49  walls[nextCell] |= (1 << SOUTH);
50  break;
51  case EAST:
52  walls[cell] |= (1 << EAST);
53  walls[nextCell] |= (1 << WEST);
54  break;
55  case SOUTH:
56  walls[cell] |= (1 << SOUTH);
57  walls[nextCell] |= (1 << NORTH);
58  break;
59  case WEST:
60  walls[cell] |= (1 << WEST);
61  walls[nextCell] |= (1 << EAST);
62  break;
63  default:; // do nothing -although this is an error
64  break;
65  }
66 }

102 /***
103  * Initialise a maze and the costs with border walls and the start cell
104  *
105  * If a test maze is provided, the walls will all be set up from that
106  * No attempt is made to verufy the correctness of a test maze.
107  *
108  */
109 void mazeInit(const unsigned char *testMaze) {
110  for (int i = 0; i < 256; i++) {
111  cost[i] = 0;
112  walls[i] = 0;
113  }
114  if (testMaze) {
115  mazeCopyWallsFromFlash(testMaze);
116  return;
117  }
118  for (unsigned char i = 0; i < 16; i++) {
119  mazeSetWall(i, WEST);
120  mazeSetWall(15 * 16 + i, EAST);
121  mazeSetWall(i * 16, SOUTH);
122  mazeSetWall((16 * i + 15), NORTH);
123  }
124  mazeSetWall(0, EAST);
125  mazeClearWall(0, NORTH);
126 }

170 /***
171  * Assumes the maze has been flooded
172  */
173 unsigned char neighbourCost(unsigned char cell, unsigned char direction) {
174  unsigned char result = MAX_COST;
175  unsigned char wallData = walls[cell];
176  switch (direction) {
177  case NORTH:
178  if ((wallData & (1 << NORTH)) == 0) {
179  result = cost[cellNorth(cell)];
180  }
181  break;
182  case EAST:
183  if ((wallData & (1 << EAST)) == 0) {
184  result = cost[cellEast(cell)];
185  }
186  break;
187  case SOUTH:
188  if ((wallData & (1 << SOUTH)) == 0) {
189  result = cost[cellSouth(cell)];
190  }
191  break;
192  case WEST:
193  if ((wallData & (1 << WEST)) == 0) {
194  result = cost[cellWest(cell)];
195  }
196  break;
197  default:
198  // TODO: this is an error. We should handle it.
199  break;
200  }

206 /***
207  * Very simple cell counting flood fills cost array with the
208  * manhattan distance from every cell to the target.
209  *
210  * Although the queue looks complicated, this is a fast flood that
211  * examines each accessible cell exactly once. Consequently, it runs
212  * in fairly constant time, taking 5.3ms when there are no interrupts.
213  *
214  * @param target - the cell from which all distances are calculated
215  */
216 void mazeFlood(unsigned char target) {
217  for (int i = 0; i < 256; i++) {
218  cost[i] = MAX_COST;
219  }
220  Queue<unsigned char> queue;
221  cost[target] = 0;
222  queue.add(target);
223  while (queue.size() > 0) {
224  unsigned char here = queue.head();
225  unsigned int newCost = cost[here] + 1;
226 
227  for (unsigned char direction = 0; direction < 4; direction++) {
228  if (hasExit(here, direction)) {
229  unsigned int nextCell = neighbour(here, direction);
230  if (cost[nextCell] > newCost) {
231  cost[nextCell] = newCost;
232  queue.add(nextCell);
233  }
234  }
235  }
236  }
237 }

240 /***
241  * Algorithm looks around the current cell and records the smallest
242  * neighbour and its direction. By starting with the supplied direction,
243  * then looking right, then left, the result will preferentially be
244  * ahead if there are multiple neighbours with the same cost.
245  *
246  * @param cell
247  * @param startDirection
248  * @return
249  */
250 unsigned char directionToSmallest(unsigned char cell, unsigned char startDirection) {
251  unsigned char nextDirection = startDirection;
252  unsigned char smallestDirection = INVALID_DIRECTION;
253  unsigned int nextCost;
254  unsigned int smallestCost = cost[cell];
255  nextCost = neighbourCost(cell, nextDirection);
256  if (nextCost < smallestCost) {
257  smallestCost = nextCost;
258  smallestDirection = nextDirection;
259  };
260  nextDirection = (startDirection + 1) % 4; // right
261  nextCost = neighbourCost(cell, nextDirection);
262  if (nextCost < smallestCost) {
263  smallestCost = nextCost;
264  smallestDirection = nextDirection;
265  };
266  nextDirection = (startDirection + 3) % 4; // left
267  nextCost = neighbourCost(cell, nextDirection);
268  if (nextCost < smallestCost) {
269  smallestCost = nextCost;
270  smallestDirection = nextDirection;
271  };
272  nextDirection = (startDirection + 2) % 4; // behind
273  nextCost = neighbourCost(cell, nextDirection);
274  if (nextCost < smallestCost) {
275  smallestCost = nextCost;
276  smallestDirection = nextDirection;
277  };
278  if (smallestCost == MAX_COST) {
279  smallestDirection = 0;
280  }
281  return smallestDirection;
282 }
