111 /***
112  * The mouse is assumed to be centrally placed in a cell and may be
113  * stationary. The current location is known and need not be any cell
114  * in particular.
115  *
116  * The walls for the current location are assumed to be correct in
117  * the map.
118  *
119  * On execution, the mouse will search the maze until it reaches the
120  * given target.
121  *
122  * The maze is mapped as each cell is entered. Mapping happens even in
123  * cells that have already been visited. Walls are only ever added, not
124  * removed.
125  *
126  * It is possible for the mapping process to make the mouse think it
127  * is walled in with no route to the target.
128  *
129  * Returns 0 if the search is successful
130  * -1 if the maze has no route to the target.
131  */
132 int mouseSearchTo(int target) {
133  mazeFlood(target);
134  mouseShowStatus();
135  debug << F(" searching to: ") << target << endl;
136  if (cost[mouse.location] == MAX_COST) {
137  return -1;
138  }
139  unsigned char newHeading;
140  if (mouse.handStart) { // implies that the heading is correct
141  mouse.handStart = false;
142  // move to the cell centre
143  forward(MM(40), SPEEDMAX_EXPLORE, SPEEDMAX_EXPLORE);
144  } else {
145  newHeading = directionToSmallest(mouse.location, mouse.heading);
146  mouseTurnToFace(newHeading);
147  }
148  while (mouse.location != target) {
149  // here the mouse is always at the center of the cell and may be
150  // stationary or moving
151  steeringMode = SM_STRAIGHT;
152  startForward(SPEEDMAX_EXPLORE);
153  motorsWaitUntil(MM(90));
154  // now we are at the cell boundary
155  mouse.location = neighbour(mouse.location, mouse.heading);
156  mouseCheckWallSensors();
157  mouseShowStatus();
158  mouseUpdateMapFromSensors();
159  mazeFlood(target);
160  if (mouse.location == target) {
161  stopAndAdjust();
162  break;
163  }
164  if (cost[mouse.location] == MAX_COST) { // are we walled in
165  stopAndAdjust();
166  return -1;
167  }
168  newHeading = directionToSmallest(mouse.location, mouse.heading);
169  unsigned char hdgChange = (newHeading - mouse.heading) & 0x3;
170  switch (hdgChange) {
171  case 0: // ahead
172  motorsWaitUntil(MM(180));
173  break;
174  case 1: // right
175  stopAndAdjust();
176  turnIP90R();
177  break;
178  case 2: // behind
179  stopAndAdjust();
180  turnIP180();
181  break;
182  case 3: // left
183  stopAndAdjust();
184  turnIP90L();
185  break;
186  }
187  }
188  return 0;
189 }

399 /***
400  * The mouse is expected to be in the start cell heading NORTH
401  * The maze may, or may not, have been searched.
402  * There may, or may not, be a solution.
403  *
404  * This simple searcher will just search to goal, turn around and
405  * search back to the start. At that point there will be a route
406  * but it is unlikely to be optimal.
407  *
408  * the mouse can run this route by creating a path that does not
409  * pass through unvisited cells.
410  *
411  * A better searcher will continue until a path generated through all
412  * cells, regardless of visited state, does not pass through any
413  * unvisited cells.
414  *
415  * The walls can be saved to EEPROM after each pass. It left to the
416  * reader as an exercise to do something useful with that.
417  */

436  }

442 /***
443  * Search the maze until there is a solution then make a path and run it
444  * First with in-place turns, then with smooth turns;
445  *
446  * The mouse can be placed into any of the possible stated before
447  * calling this function so that individual actions can be tested.
448  *
449  * If you do not want to search exhaustively then do a single search
450  * out and back again. Then block off all the walls in any cells that
451  * are unvisited. Now any path generated will succeed even if it is
452  * not optimal.
453  */
454 int mouseRunMaze() {
455  motorsEnable();
456  if (mouseState == SEARCHING) {
457  if (waitForStart() == 0) {
458  return 0;
459  }
460  mouse.handStart = true;
461  steeringMode = SM_STRAIGHT;
462  mouse.location = 0;
463  mouse.heading = NORTH;
464  mouseSearchTo(GOAL);
465  digitalWrite(GREEN_LED, 1);
466  mouseSearchTo(0);
467  digitalWrite(GREEN_LED, 0);
468  digitalWrite(RED_LED, 1);
469  mouseTurnToFace(NORTH);
470  // we have a solution and the mouse is at the start ready to run
471  digitalWrite(RED_LED, 0);
472  delay(200);
473  mouseState = INPLACE_RUN;
474  }
475  if (mouseState == INPLACE_RUN) {
476  mazeFlood(GOAL);
477  pathGenerate(0);
478  debug << F("Maze is searched\nwaiting inplace for start\n");
479  if (waitForStart() == 0) {
480  return 0;
481  }
482  digitalWrite(GREEN_LED, 0);
483  digitalWrite(RED_LED, 0);
484  console.println(F("Running in place"));
485  mouseRunInplaceTurns(SPEEDMAX_STRAIGHT);
486  console.println(F("Returning"));
487  mouseSearchTo(0);
488  console.println(F("Done"));
489  mouseState = SMOOTH_RUN;
490  }
491  if (mouseState == SMOOTH_RUN) {
492  // now try with smooth turns;
493  mazeFlood(GOAL);
494  pathGenerate(0);
495  mouseTurnToFace(directionToSmallest(mouse.location, mouse.heading));
496  delay(200);
497  debug << F("waiting for smooth run start\n");
498  if (waitForStart() == 0) {
499  return 0;
500  }
501  console.println(F("Running smooth"));
502  mouseRunSmoothTurns(SPEEDMAX_STRAIGHT);
503  console.println(F("Returning"));
504  mouseSearchTo(0);
505  console.println(F("Finished"));
506  mouseState = FINISHED;
507  }
508  motorsDisable();
509  return 0;
510 }

514 /***
515  * Assumes the maze is already flooded to a single target cell and so
516  * every cell will have a cost that decreases as the target is approached.
517  *
518  * Starting at the given cell, the algorithm repeatedly looks for the
519  * smallest available neighbour and records the action taken to reach it.
520  *
521  * The process starts by assuming the mouse is heading NORTH in the start
522  * cell since that is what would be the case at the start of a speed run.
523  *
524  * At each cell, the preference is to move forwards if possible
525  *
526  * If the pathfinder is called from any other cell, the mouse must first
527  * turn to face to the smallest neighbour of that cell using the same
528  * method as in this function.
529  *
530  * The resulting path is a simple string, null terminated, that can be
531  * printed to the console to make it easy to compare paths using different
532  * flooding or path generating methods.
533  *
534  * The characters in the path string are:
535  * 'B' : always the first character, it marks the path start.
536  * 'F' : move forwards a full cell
537  * 'H' : used in speedruns to indicate movement of half a cell forwards
538  * 'R' : turn right in this cell
539  * 'L' : turn left in this cell
540  * 'A' : turn around (should never happen in a speedrun path)
541  * 'S' : the last character in the path, telling the mouse to stop
542  *
543  * For example, the Japan2007 maze, flooded with a simple Manhattan
544  * flood, should produce the path string:
545  *
546  * BFFFRLLRRLLRRLLRFFRRFLLFFLRFRRLLRRLLRFFFFFFFFFRFFFFFRLRLLRRLLRRFFRFFFLFFFS
547  *
548  * The path string is processed by the mouse directly to make it move
549  * along the path. At its simplest, this is just a case of executing
550  * a single movement for each character in the string, using in-place turns.
551  *
552  * I would strongly recommend this style of path string. Not only can the
553  * strings be used to compare routes very easily, they can be printed and
554  * visually compared or followed by hand.
555  *
556  * Path strings are easily translated into more complex paths using
557  * smooth turns an they are relatively easy to turn into a set of
558  * commands that will represent a diagonal path.
559  *
560  * Further, short path strings can be hand-generated to test the movement
561  * of the mouse or to test the setup of different turn types.
562  *
563  * The pathGenerator is not terribly efficient and can take up to 20ms to
564  * generate a path depending on the maze, start cell and target.
565  *
566  */
567 
568 bool pathGenerate(unsigned char startCell) {
569  bool solved = true;;
570  unsigned char cell = startCell;
571  int nextCost = cost[cell] - 1; // assumes manhattan flood
572  unsigned char commandIndex = 0;
573  path[commandIndex++] = 'B';
574  unsigned char direction = directionToSmallest(cell, NORTH);
575  while (nextCost >= 0) {
576  unsigned char cmd = 'S';
577  switch (direction) {
578  case NORTH:
579  if (hasExit(cell, NORTH) && neighbourCost(cell, NORTH) == nextCost) {
580  cmd = 'F';
581  break;
582  }
583  if (hasExit(cell, EAST) && neighbourCost(cell, EAST) == nextCost) {
584  cmd = 'R';
585  direction = DtoR[direction];
586  break;
587  }
588  if (hasExit(cell, WEST) && neighbourCost(cell, WEST) == nextCost) {
589  cmd = 'L';
590  direction = DtoL[direction];
591  break;
592  }
593  if (hasExit(cell, SOUTH)
594  && neighbourCost(cell, SOUTH) == nextCost) {
595  cmd = 'A';
596  direction = DtoB[direction];
597  }
598  break;
599  case EAST:
600  if (hasExit(cell, EAST) && neighbourCost(cell, EAST) == nextCost) {
601  cmd = 'F';
602  break;
603  }
604  if (hasExit(cell, SOUTH)
605  && neighbourCost(cell, SOUTH) == nextCost) {
606  cmd = 'R';
607  direction = DtoR[direction];
608  break;
609  }
610  if (hasExit(cell, NORTH)
611  && neighbourCost(cell, NORTH) == nextCost) {
612  cmd = 'L';
613  direction = DtoL[direction];
614  break;
615  }
616  if (hasExit(cell, WEST) && neighbourCost(cell, WEST) == nextCost) {
617  cmd = 'A';
618  direction = DtoB[direction];
619  break;
620  }
621  break;
622  case SOUTH:
623  if (hasExit(cell, SOUTH)
624  && neighbourCost(cell, SOUTH) == nextCost) {
625  cmd = 'F';
626  break;
627  }
628  if (hasExit(cell, WEST) && neighbourCost(cell, WEST) == nextCost) {
629  direction = DtoR[direction];
630  cmd = 'R';
631  break;
632  }
633  if (hasExit(cell, EAST) && neighbourCost(cell, EAST) == nextCost) {
634  cmd = 'L';
635  direction = DtoL[direction];
636  break;
637  }
638  if (hasExit(cell, NORTH)
639  && neighbourCost(cell, NORTH) == nextCost) {
640  cmd = 'A';
641  direction = DtoB[direction];
642  break;
643  }
644  break;
645  case WEST:
646  if (hasExit(cell, WEST) && neighbourCost(cell, WEST) == nextCost) {
647  cmd = 'F';
648  break;
649  }
650  if (hasExit(cell, NORTH) && neighbourCost(cell, NORTH) == nextCost) {
651  cmd = 'R';
652  direction = DtoR[direction];
653  break;
654  }
655  if (hasExit(cell, SOUTH)
656  && neighbourCost(cell, SOUTH) == nextCost) {
657  cmd = 'L';
658  direction = DtoL[direction];
659  break;
660  }
661  if (hasExit(cell, EAST) && neighbourCost(cell, EAST) == nextCost) {
662  cmd = 'A';
663  direction = DtoB[direction];
664  break;
665  }
666  break;
667  default:
668  // this is an error. We should handle it.
669  break;
670  }
671  cell = neighbour(cell, direction);
672  if ((walls[cell] & VISITED) != VISITED) {
673  solved = false;
674  }
675  nextCost--;
676  path[commandIndex] = cmd;
677  commandIndex++;
678  }
679  path[commandIndex] = 'S';
680  commandIndex++;
681  path[commandIndex] = '\0';
682  return solved;
683 }

685 /***
686  * Assumes that the maze is flooded and a path string has been generated.
687  *
688  * Convert the simple path string to a set of commands using half-cell
689  * moves instead of full-cell moves so that a sequence like
690  * BFRFLS
691  * becomes
692  * BHHRHHHHLHHS
693  *
694  * The turns all have an implied full cell forward movement after them
695  * The only real advantage of this is that it is easier to convert to smooth
696  * turns by looking for patterns like HRH and HLH
697  *
698 
699  */
700 void pathExpand(char * pathString) {
701  int pathIndex = 0;
702  int commandIndex = 0;
703  commands[commandIndex++] = 'B';
704  while (char c = pathString[pathIndex]) {
705  switch (c) {
706  case 'F':
707  commands[commandIndex++] = 'H';
708  commands[commandIndex++] = 'H';
709  pathIndex++;
710  break;
711  case 'R':
712  commands[commandIndex++] = 'R';
713  commands[commandIndex++] = 'H';
714  commands[commandIndex++] = 'H';
715  pathIndex++;
716  break;
717  case 'L':
718  commands[commandIndex++] = 'L';
719  commands[commandIndex++] = 'H';
720  commands[commandIndex++] = 'H';
721  pathIndex++;
722  break;
723  case 'S':
724  commands[commandIndex++] = 'S';
725  pathIndex++;
726  break;
727  case 'B':
728  case ' ': // ignore these so we can write easy-to-read path strings
729  pathIndex++;
