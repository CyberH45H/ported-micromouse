156 // Peter Harrison - Jan 2018
157 // Allow an integer to be right aligned in a field of specified width
158 // no attempt is made to guard against overflow - the field will overflow
159 // Serial << _JUSTIFY(-1234,8) << endl;

163  int width;
164  _JUSTIFY(int v, int w) : val(v), width(w) {};
165 };
166 
167 inline Print &operator <<(Print &obj, const _JUSTIFY &arg) {
168  int v = arg.val;
169  int w = arg.width;
170  w--;
171  if (v < 0) {
172  w--;
173  }
174  while (v /= 10) {
175  w--;
176  }
177  while (w > 0) {
178  obj.write(' ');
179  --w;
180  }
181  obj.print(arg.val);
182  return obj;
183 }
