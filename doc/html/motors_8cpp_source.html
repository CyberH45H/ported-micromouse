69 void motorRightUpdate() {
70 
71  unsigned int timerInterval;
72 
73  if (speedRight < speedTargetRight) {
74  speedRight++;
75  }
76  if (speedRight > speedTargetRight) {
77  speedRight--;
78  }
79  speedRight = constrain(speedRight, 0, SPEED_TABLE_END);
80  if (speedRight == 0) {
81  timerInterval = MOTOR_IDLE_57Hz;
82  } else {
83  digitalWriteFast(STEPR, 1);
84  timerInterval = accTable(speedRight);
85  if (slowRightMotor) {
86  timerInterval += (timerInterval / 8);
87  }
88  offsetCount++;
89  positionCount++;
90  digitalWriteFast(STEPR, 0);
91  }
92  OCR1B += timerInterval;
93 }

168 void motorsStopAt(long target) {
169  bool done = false;
170  while (!done) {
171  int remaining = target - getStepCount();
172  int speed = getVolatile(speedRight) + getVolatile(speedLeft);
173  if (remaining < speed) {
174  done = true;
175  }
176  }
177  // hit the brakes
178  setVolatile(speedTargetRight, 1);
179  setVolatile(speedTargetLeft, 1);
180  done = false;
181  while (!done) {
182  done = getStepCount() >= target;
183  }
184  // make sure they stop completely
185  setVolatile(speedTargetRight, 0);
186  setVolatile(speedTargetLeft, 0);
187  setVolatile(speedRight, 0);
188  setVolatile(speedLeft, 0);
189 }

193 /*
194  * microstepping modes for DRV8834. Z means pin is left floating
195  *
196  * M1 M0 Mode
197  * 0 0 Full step (1)
198  * 0 1 half step (2)
199  * 0 Z quarter (4)
200  * 1 0 8th (8)
201  * 1 1 16th (16)
202  * 1 Z 32nd (32)
203  */
204 void setMicrostepMode(int mode) {
205  switch (mode) {
206  case MICROSTEP_1:
207  pinMode(M1, OUTPUT);
208  pinMode(M0, OUTPUT);
209  digitalWriteFast(M1, 0);
210  digitalWriteFast(M0, 0);
211  break;
212  case MICROSTEP_2:
213  pinMode(M1, OUTPUT);
214  pinMode(M0, OUTPUT);
215  digitalWriteFast(M1, 0);
216  digitalWriteFast(M0, 1);
217  break;
218  case MICROSTEP_4:
219  pinMode(M1, OUTPUT);
220  pinMode(M0, INPUT);
221  digitalWriteFast(M1, 0);
222  digitalWriteFast(M0, 0);
223  break;
224  case MICROSTEP_8:
225  pinMode(M1, OUTPUT);
226  pinMode(M0, OUTPUT);
227  digitalWriteFast(M1, 1);
228  digitalWriteFast(M0, 0);
229  break;
230  case MICROSTEP_16:
231  pinMode(M1, OUTPUT);
232  pinMode(M0, OUTPUT);
233  digitalWriteFast(M1, 1);
234  digitalWriteFast(M0, 1);
235  break;
236  case MICROSTEP_32:
237  pinMode(M1, OUTPUT);
238  pinMode(M0, INPUT);
239  digitalWriteFast(M1, 1);
240  digitalWriteFast(M0, 0);
241  break;
242  default:
243  // TODO: this is an error. We should handle it.
244  break;
245  }
