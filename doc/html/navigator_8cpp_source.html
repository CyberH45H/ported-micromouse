42 /***
43  *
44  * getSteeringError() is called by the navigator to allow correction of
45  * alignment (heading) errors in the mouse while moving in a straight
46  * line.
47  *
48  * The way the steering error is calculated depends upon the current mode.
49  *
50  * Normally the side sensors are used and the method is simply a question
51  * of collision avoidance. the mouse is prevented from getting too close
52  * to a wall. This can be a problem if there is a wall down just one side
53  * of a long straight since the mouse may steer away and find no useful
54  * correction from the other side.
55  *
56  * The front sensors can help since, with normal alignment, they cannot
57  * see the walls or posts at the sides of the mouse. However, they are
58  * aligned so that a heading or offset error will begin to generate a
59  * response. That response will override the side sensor readings.
60  *
61  * Since the responses from posts alone are short lived transients
62  * they may not last long enough to significantly affect the steering.
63  * A low pass filter on the error signal prevents sudden step responses
64  * and effectively prolongs the effect of a single post that is too close.
65  * This low pass filter will need tuning by adjusting the value of alpha.
66  * Alpha should probably be in the mouse parameter header.
67  *
68  * When the mouse is too close to a wall ahead, that wall will affect the
69  * side sensor readings. To prevent that, a check should be made on the front
70  * sensor values and when that condition is met, the steering mode should
71  * be switched to SM_FRONT where the difference in the front sensor values
72  * can be used for alignment.
73  *
74  * NOTE that the sensor responses are non-linear and likely to be quite
75  * different from side to side unless they are very carefully set up.
76  *
77  * This will show up in the observation that a lateral error of the same
78  * amount left and right will give different error values and so different
79  * responses.
80  *
81  * Don't try and fix this in software. Get the sensor alignment right so
82  * that the left and right responses are as close to the same as possible.
83  *
84  */
85 
86 int sideSensorError() {
87  int left = sensL;
88  int right = sensR;
89  int error = 0;
90  if (left > LD_NOMINAL && right > RD_NOMINAL) {
91  error = left - right;
92  } else if (left < right && right > RD_NOMINAL) {
93  error = RD_NOMINAL - right;
94  } else if (left > right && left > LD_NOMINAL) {
95  error = left - LD_NOMINAL;
96  }
97  return error;
98 }

100 int getSteeringError() {
101  int error = 0;
102  static int errorOld;
103  switch (steeringMode) {
104  case SM_NONE:
105  error = 0;
106  break;
107  case SM_STRAIGHT:
108  error = sideSensorError();
109  break;
110  case SM_FRONT:
111  // in this mode, the mouse should work out the distance to
112  // the wall ahead and its orientation.
113  // From these, the steering error can be calculated
114  if ((sensFL + sensFR) > FRONT_WALL_INTERFERENCE_THRESHOLD) {
115  // add code to use the front sensor readings
116  error = 0;
117  } else {
118  error = sideSensorError();
119  }
120  break;
121  default:
122  error = 0;
123  break;
124  }
125  /***
126  * low pass filter the error value to get a smooth response to steps
127  * Be sure that there can be no overflow here. It should be safe since
128  * the error cannot exceed 900 except if there is a fault.
129  * And, yes, there are rounding errors here. They are acceptable.
130  */
131  const int alpha = 28;
132  error = (alpha * errorOld + (32 - alpha) * error) / 32;
133  errorOld = error;
134  return error;
135 }

212 /***
213  * steering alignment is very simple and probably only suited to a
214  * stepper design. Ideally it should be proportional and it should
215  * work by applying a rotational component to the motors. However,
216  * in a stepper mouse, it may simply not be possible to drive one
217  * motor any faster so the rotation component would get complicated
218  * to manage.
219  *
220  * Instead, here we just slow down one of the motors to make the
221  * mouse turn. A fixed ratio is used for the change and the duration
222  * of the correction is proportional to the error. In effect, that
223  * makes the correction an angle and alignment errors are assumed
224  * to be angular errors.
225  *
226  * Experiment with the amount and duration of the correction to tune
227  * the mouse steering.
228  */
229 void doAlignment() {
230  // always update the steering error even if it is not used so that
231  // it can be observed and logged
232  // positive values mean we need to turn right
233  steeringError = constrain(getSteeringError(), -STEERING_ERROR_MAX, STEERING_ERROR_MAX);
234 
235  if (alignmentCounter > 0) { // there is already a correction taking place
236  alignmentCounter--;
237  return;
238  }
239 
240  // assume no correction is needed
241  alignmentCounter = 0;
242  slowLeftMotor = 0;
243  slowRightMotor = 0;
244 
245  if (steeringError > 0) {
246  slowRightMotor = 1;
247  // alignmentCounter = steeringError;
248  }
249  if (steeringError < 0) {
250  slowLeftMotor = 1;
251  // alignmentCounter = -steeringError;
252  }
253 }
