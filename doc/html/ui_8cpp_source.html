104 /***
105  * Debouncing here is a simple state machine for one pin. The debounce time is in
106  * system ticks For the PORTED mouse these are at 250Hz. The pin must stay in the
107  * same state for two states so, for a 24ms debounce time, the number of debounceTicks
108  * will be 3 per state
109  *
110  * For good quality buttons, small debounce times will probably be reliable and will give
111  * good, short response times. Poor quality switches or mechanical sensors like bumpers
112  * are likely to require longer debounce times but that will make them less responsive.
113  *
114  * The pin state can be active high or low and defaults to active low. This would be a pin
115  * that is pulled up by a resistor and shorted to ground when the button is pressed.
116  *
117  * The pressed() and released() functions use read() to get the instantaneous state of
118  * the pin.
119  *
120  * Note that pressed() and released() are not logical opposites since, at the time
121  * they get called, the button may be changing state. It is not safe to substitute
122  * pressed(BUTTON) with !released(BUTTON).
123  *
124  * However, for simple use, it is generally acceptable use pressed() and !pressed().
125  *
126  * To make PRESSED and RELEASED the only two exported states would require persistent
127  * storage of state for the button and this is not needed for a simple, one button solution.
128  *
129  * In normal use the user code might call pressed() to test a pin state and act if it
130  * returns true. This is how buttons most commonly work:
131  *
132  * if (pressed(BUTTON)){
133  * callFunction();
134  * while(!released(BUTTON)
135  * }
136  *
137  * However, for something like starting a mouse in the maze or for a test, it is not a good
138  * idea to have the action begin until the button is released. That can be achieved like this:
139  *
140  * if (pressed(BUTTON)){
141  * while(!released(BUTTON)){};
142  * callFunction();
143  * }
144  *
145  * Or, if more convenient:
146  *
147  * waitForClick(BUTTON);
148  *
149  *
150  */

161 void debouncePin(unsigned char pin) {
162  static unsigned char ticks = 0;
163  if (++ticks < debounceTicks) {
164  return;
165  }
166  ticks = 0;
167  switch (pinState) {
168  case RELEASED:
169  if (read(pin)) {
170  pinState = PRESSING;
171  } else {
172  pinState = RELEASED;
173  }
174  break;
175  case PRESSING:
176  if (read(pin)) {
177  pinState = PRESSED;
178  } else {
179  pinState = RELEASED;
180  }
181  break;
182  case PRESSED:
183  if (read(pin)) {
184  pinState = PRESSED;
185  } else {
186  pinState = RELEASING;
187  }
188  break;
189  case RELEASING:
190  if (read(pin)) {
191  pinState = PRESSED;
192  } else {
193  pinState = RELEASED;
194  }
195  break;
196  }
197 }
