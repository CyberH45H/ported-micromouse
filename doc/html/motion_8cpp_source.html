38 /***
39  * The integrated smooth turn simply speeds up the outer wheel and
40  * slows down the inner wheel. The speeds are not specified as such
41  * because of the way the stepper tables work. Instead, the wheels
42  * accelerate to their new speeds over a fixed distance. The actual
43  * wheel speeds after this are then held steady for another fixed
44  * distance and the wheels are then allowed to return to their initial
45  * speeds - again over a fixed distance.
46  *
47  * In this way, the key turn parameters are just the entry speed and
48  * two distances. Only two because the first and last phase are the same.
49  *
50  * combined with the entry speed, the phase 1 and phase 3 distances will
51  * determine the minimum radius of the turn. The phase 2 distance can be
52  * adjusted to change the total angle turned by the mouse.
53  *
54  * This method is simple and effective but there is no obvious analytical
55  * solution for a turn of a given speed and radius. It can be worked out
56  * but that is left as an exercise to the reader.
57  *
58  * When setting up the straight and turn speeds, care must be taken to
59  * ensure that the mouse can comfortably get down to the speed used when
60  * setting up the phase1 and phase2 distances.
61  *
62  * The turning parameters - speeds, angles and distances - should really
63  * be held in a table so that it is easy to look up the values
64  * needed for different turn types. For example, rather than do two
65  * successive right angle turns, it is better to do one smooth 180
66  * degree turn.
67  *
68  * For this, the path processing will need more work and turnSmooth will
69  * probably need to be given a turn type and direction.
70  *
71  * In a DC mouse, a very similar process is used. There are still the three
72  * phases but the desired angular velocity is easily calculated in advance
73  * given knowledge of the entry speed and the desired radius. Note that
74  * the relation ship V = w x R always holds. R is the radius, V is the
75  * tangential speed and w is the angular velocity. Thus, to run a turn
76  * faster but with the same radius, w must be increased in proportion.
77  */
78 void turnSmooth(int direction) {
79  steeringMode = SM_NONE;
80  motorsSetDirection(FORWARD);
81  int phase1 = 220; // adjust for radius
82  int phase2 = 500; // adjust for angle
83  noInterrupts();
84  if (direction == RIGHT) {
85  speedTargetLeft = 800;
86  speedTargetRight = 1;
87  } else {
88  speedTargetLeft = 1;
89  speedTargetRight = 800;
90  }
91  positionCount = 0;
92  long targetSteps = phase1;
93  interrupts();
94  // phase 1 - accelerating phase
95  while (getStepCount() < targetSteps) {
96  ; // do nothing
97  }
98  // phase 2 - constant radius
99  noInterrupts();
100  speedTargetLeft = speedLeft;
101  speedTargetRight = speedRight;
102  targetSteps += phase2;
103  interrupts();
104  while (getStepCount() < targetSteps) {
105  ; // do nothing
106  }
107  // phase 3 - decelerating phase
108  noInterrupts();
109  speedTargetLeft = SPEEDMAX_SMOOTH_TURN;
110  speedTargetRight = SPEEDMAX_SMOOTH_TURN;
111  targetSteps += phase1;
112  interrupts();
113  while (getStepCount() < targetSteps) {
114  ; // do nothing
115  }
116  noInterrupts();
117  speedLeft = SPEEDMAX_SMOOTH_TURN;
118  speedRight = SPEEDMAX_SMOOTH_TURN;
119  interrupts();
120 }

124 /***
125  * General purpose motion profiler where both motors have to move at the same speed.
126  * Used for forward/reverse moves and in-place turns;
127  *
128  * Profiling is a simple question of accelerating the mouse until we need
129  * to brake while limiting the maximum speed.
130  *
131  * For a stepper mouse, this is easy because the number of steps needed
132  * to bring a motor to a given speed index is just the difference in index
133  * values for the two speeds.
134  *
135  * Private to this module. Only call from other code that sets up the motor
136  * directions correctly
137  */
138 void move(long steps, int maxSpeed, int exitSpeed) {
139  noInterrupts();
140  speedTargetLeft = maxSpeed;
141  speedTargetRight = maxSpeed;
142  positionCount = 0;
143  interrupts();
144  long brakingSteps;
145  long remainingSteps;
146  // accelerating phase
147  do {
148  remainingSteps = steps - getVolatile(positionCount);
149  brakingSteps = getVolatile(speedRight) + getVolatile(speedLeft);
150  // NOTE: do not divide speed by 2 as we want the total number of
151  // braking steps here
152  brakingSteps -= exitSpeed * 2;
153  } while (remainingSteps >= brakingSteps);
154  // decelerating phase
155  if (exitSpeed > 0) {
156  setVolatile(speedTargetRight, exitSpeed);
157  setVolatile(speedTargetLeft, exitSpeed);
158  } else { // ensure we get to that last step;
159  setVolatile(speedTargetRight, 1);
160  setVolatile(speedTargetLeft, 1);
161  }
162  while (getVolatile(positionCount) < steps) {
163  ; // do nothing
164  }
165  // force the current speed to match the set speed
166  setVolatile(speedTargetRight, exitSpeed);
167  setVolatile(speedTargetLeft, exitSpeed);
168  setVolatile(speedRight, exitSpeed);
169  setVolatile(speedLeft, exitSpeed);
170 }

172 /***
173  * Turn in place by the given number of steps.
174  * Steps will be the sum of the left and right motor step counts
175  * Positive steps is anticlockwise or left, negative is clockwise or right
176  * maxSpeed and exitSpeed are indexes into the acceleration tables.
177  * Normally, the exit speed would be zero but it need not be.
178  */
179 void spin(long steps, int maxSpeed, int exitSpeed) {
180  steeringMode = SM_NONE;
181  if (steps >= 0) {
182  motorsSetDirection(LEFT);
183  } else {
184  motorsSetDirection(RIGHT);
185  steps = -steps;
186  }
187  move(steps, maxSpeed, exitSpeed);
188 }
